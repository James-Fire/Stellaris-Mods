upload_planet_1 = {
	sound = event_administrative_work
	icon = decision_resources
	
	prerequisites = {
		"tech_mass_drivers_1"
	}
	
	resources = {
		category = decisions
		cost = {
			rare_crystals = 500
		}
	}
	enactment_time = 900
	potential = {
		OR = {
			exists = owner
			owner = {
				has_origin = origin_transcendent
				has_technology = tech_synthetic_workers
			}
		}
		NOR = {
			has_modifier = planet_uploaded_1
			has_modifier = planet_uploaded_2
		}
		is_colony = yes
		NOR = {
			is_planet_class = pc_habitat
			is_planet_class = pc_virtual_reality
			is_planet_class = pc_virtual_industry
			is_planet_class = pc_virtual_stellar_industry
		}
	}
	
	allow = {
		always = yes
	}	
	
	effect = {
		add_modifier = { modifier = planet_uploaded_1 days = -1 }
	}
	
	ai_weight = {
		weight = 0
	}
}
