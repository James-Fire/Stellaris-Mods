namespace = est_cv

#Industry 2, gives planet mineral deposits.
planet_event = {
	id = est_cv.3
	hide_window = yes
	is_triggered_only = yes
		
	trigger = {
		owner = {
			has_tradition = tr_est_industry_2
		}
	}
	
	immediate = {
		if = {
			limit = {
				Not = {
					has_planet_flag = industry_2_flag
					is_planet_class = pc_habitat 
					has_modifier = mod_esap_ecumenopolis
				}
			}
			set_planet_flag = industry_2_flag
			every_tile = {
				random_list = {
					3 = { }
					1 = {
						add_deposit = d_rich_mineral_deposit
					}
				}
			}
		}
	}
}

#Liberty 4, bonus while fighting the bad guys.
event = {
	id = est_cv.5
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {	
		any_country = {
			has_tradition = tr_est_liberty_4
		}	
	}
	
	immediate = {
		every_country = {
			limit = {
				has_tradition = tr_est_liberty_4			
			}
			if = {
				limit = {
					any_war = {
						any_war_participant = {
							is_war_leader = yes
							is_at_war_with = prevprev
							Or = {
								has_policy_flag = purge_allowed
								has_policy_flag = purge_displacement_only
								has_policy_flag = slavery_allowed
								has_valid_civic = civic_fanatic_purifiers
							}							
						}
					}							
				}
				if = {
					limit = {
						Not = {
							has_modifier = mod_est_cv_liberty_4
						}
					}
					add_modifier = {
						modifier = "mod_est_cv_liberty_4"
						days = -1
					}
				}				
				else = {
					if = {
						limit = {
							has_modifier = mod_est_cv_liberty_4
						}
						remove_modifier = "mod_est_cv_liberty_4"
					}
				}
			}
		}
	}
}

#Purity adopt, bonus on worlds without other species.
event = {
	id = est_cv.7
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		any_country = {
			has_tradition = tr_est_purity_adopt
		}
	}
	
	## Vad händer när sådan planet errövras?
	immediate = { 
		every_country = {
			limit =  {
				has_tradition = tr_est_purity_adopt
			}
			every_owned_planet = {
				if = {
					limit = {
						Not = {
							any_owned_pop = {
								Not = {
									is_same_species = owner
								}
							}
						}
					}
					if = {
						limit = {
							Not = {
								has_modifier = mod_est_cv_purity_adopt_1
							}
						}
						add_modifier = {
							modifier = "mod_est_cv_purity_adopt_1"
							days = -1
						}						
					}
					else = {
						if = {
							limit = {
								has_modifier = mod_est_cv_purity_adopt_1
							}
							remove_modifier = "mod_est_cv_purity_adopt_1"
						}
						if = {
							limit = {
								Not = {
									owner = {
										has_country_flag = got_bad_planet		 
									}
								}
							}
							owner = {
								set_country_flag = got_bad_planet							
							}
						}
					}
				}
			}
			if = {
				limit = {
					has_country_flag = got_bad_planet
				}
				remove_country_flag = got_bad_planet	
				if = {
					limit = {
						has_modifier = mod_est_cv_purity_adopt_2		
					}							
					remove_modifier = "mod_est_cv_purity_adopt_2"
				}
				else = {
					if = {
						limit = {
							Not = {
								has_modifier = mod_est_cv_purity_adopt_2
							}			
						}
						add_modifier = {
							modifier = "mod_est_cv_purity_adopt_2"
							days = -1
						}	
					}							
				}
			}
		}		
	}
}




#Commerce 1, bonus if positive energy gain.
event = {
	id = est_cv.8
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		any_country = {
			has_non_swapped_tradition = tr_est_commerce_1
		}
	}

	immediate = {
		every_country = {
			limit = {
				has_non_swapped_tradition = tr_est_commerce_1
			}	
			if = {
				limit = {
					running_balance > 0				
				}
				if = {
					limit = {
						Not = {
							has_modifier = mod_est_commerce_1
						}
					}
					add_modifier = {
						modifier = "mod_est_commerce_1"
						days = -1
					}
				}
				else = {
					if = {
						limit = {
							has_modifier = mod_est_commerce_1
						}
						remove_modifier = "mod_est_commerce_1"
					}				
				}
			}
		}
	}
}
