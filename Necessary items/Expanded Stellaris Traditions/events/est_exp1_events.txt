#Confederacy Adopt, gives bonus to sector planets.
event = {
	id = est_exp1.1
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {	
		any_country = {
			has_tradition = tr_est_confederacy_adopt
		}
	}
	
	immediate = {
		every_country = {
			limit = {
				has_tradition = tr_est_confederacy_adopt
			}
			every_owned_planet = {
				if = {
					limit = {
						sector_controlled = yes 
						Not = {
							And = {
								sector = {
									exists = leader
								}
								owner = {
									has_tradition = tr_est_confederacy_2
								}
							}
						}
					}
					if = {
						limit = {
							has_modifier = mod_confederacy_2								
						}
						remove_modifier = "mod_confederacy_2"
					}	
					if = {
						limit = {
							has_modifier = mod_subminds_2								
						}
						remove_modifier = "mod_subminds_2"
					}	
					if = {
						limit = {
							Not = {
								has_modifier = mod_confederacy_adopt
							}							
						}
						if = {
							limit = {
								owner = {
									has_non_swapped_tradition = tr_est_confederacy_adopt	
								}	
							}
							if = {
								limit = {
									Not = {
										has_modifier = mod_confederacy_adopt
									}
								}
								add_modifier = {
									modifier = "mod_confederacy_adopt"							
									days = -1
								}
							}
							else = {
								if = {
									limit = {
										Not = {
											has_modifier = mod_subminds_adopt
										}
									}
									add_modifier = {
										modifier = "mod_subminds_adopt"							
										days = -1
									}
								}
							}
						}
					}
					else = {
						if = {
							limit = {
								has_modifier = mod_confederacy_adopt								
							}
							remove_modifier = "mod_confederacy_adopt"
						}	
						if = {
							limit = {
								has_modifier = mod_subminds_adopt								
							}
							remove_modifier = "mod_subminds_adopt"
						}						
					}
				}
			}
		}
	}
}

#Confederacy 3, reduce upkeep cost for Military Stations controlled by Sectors.
event = {
	id = est_exp1.3
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {	
		any_country = {
			has_tradition = tr_est_confederacy_3
		}
	}
	
	immediate = {
		every_country = {
			limit = {
				has_tradition = tr_est_confederacy_2
			}
			every_owned_ship = {
				limit = {
					Or = {
						is_ship_size = military_station_small
						is_ship_size = military_station_medium
						is_ship_size = military_station_large
					}
				}
				if = {
					limit = {
						fleet = {
							sector_controlled = yes 		
						}
					}
					if = {
						limit = {
							Not = {
								has_modifier = mod_confederacy_3
							}
						}
						add_modifier = {
							modifier = "mod_confederacy_3"
							days = -1
						}
					}					
					else = {
						if = {
							limit = {
								has_modifier = mod_confederacy_3								
							}
							remove_modifier = "mod_confederacy_3"
						}					
					}
				}
			}
		}
	}
}

#Cybernetics 1, adds Energy bonus to robots when built. 
pop_event = {
	id = est_exp1.5
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {	
		owner = {
			has_non_swapped_tradition = tr_est_cybernetics_1		
		}
	}
	
	immediate = {
		switch = {
			trigger = pop_has_trait			
			trait_robotic_1 = { add_modifier = {modifier = "mod_cybernetics_1_robot" days = -1}}
			trait_robotic_2 = { add_modifier = {modifier = "mod_cybernetics_1_droid" days = -1} }
			trait_robotic_3 = { add_modifier = {modifier = "mod_cybernetics_1_synthetic" days = -1} }
		}
	}
}
