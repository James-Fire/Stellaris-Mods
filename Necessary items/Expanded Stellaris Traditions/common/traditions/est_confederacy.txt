@cost = 56

tr_est_confederacy_adopt = {
	cost = { unity = @cost }
	
	on_enabled = {
		every_owned_planet = {
			if = {
				limit = {
					sector_controlled = yes 
				}
				add_modifier = {
					modifier = "mod_confederacy_adopt"							
					days = -1
				}
			}
		}				
	}
	
	tradition_swap = {
		name = tr_est_subminds_adopt
		inherit_effects = no
		inherit_icon = yes
		inherit_name = no
		trigger = {
			has_ethic = ethic_gestalt_consciousness
		}	
		on_enabled = {
			every_owned_planet = {
				if = {
					limit = {
						sector_controlled = yes 
					}
					add_modifier = {
						modifier = "mod_subminds_adopt"							
						days = -1
					}
				}
			}				
		}
		weight = {
			factor = 1
		}
	}
	
	#Check event est_exp1.1
	
	ai_weight = {
		factor = 10
		modifier = {
			factor = 2
			num_owned_planets > 10 
		}	
		modifier = {
			factor = 2
			num_owned_planets > 30 
		}	
		modifier = {
			factor = 2
			num_owned_planets > 60 
		}	
	}
}
