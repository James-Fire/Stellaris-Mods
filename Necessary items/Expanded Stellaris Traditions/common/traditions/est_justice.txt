	on_enabled = {
		every_owned_pop = {
			limit = {
				Or = {
					has_citizenship_type = {
						country = owner
						type = citizenship_full
					}
					And = {
						has_citizenship_type = {
							country = owner
							type = citizenship_caste_system
						}
						is_enslaved = no				
					}
				}
			}
			add_modifier = {
				modifier = "mod_justice_finish"
				days = -1
			}		
		}
	}
	
	#Check Even test_exp1.16-20

		inherit_icon = yes
		inherit_name = no

			has_ethic = ethic_gestalt_consciousness
		}

			if = {
				limit = {
					is_at_war = yes
				}

					modifier = mod_est_destruction_finish
					days = -1
				}
			}
		}
		modifier = {
			ascension_perks_add = 1
		}
		weight = {factor = 1 }
	}

	cost = { unity = @cost }
	
	possible = {
		has_tradition = tr_est_justice_1
	}
	
	on_enabled = {
		every_pop_faction = {
			limit = {
				has_modifier = suppressed_faction
				owner = {
					Not = {
						has_country_flag = est_justice_supressing_faction
					}
				}
			}
			owner = {
				set_country_flag = est_justice_supressing_faction
			}
		}
		if = {
			limit = {
				Not = {
					has_country_flag = est_justice_supressing_faction
				}
			}
			every_pop_faction = {
				add_modifier = {
					modifier = "mod_justice_2"
					days = -1
				}
			}
		}
		if = {
			limit = {
				has_country_flag = est_justice_supressing_faction
			}
			remove_country_flag = est_justice_supressing_faction
		}
	}
	
	#Check Event
	
	tradition_swap = {
		name = tr_est_destruction_2
		inherit_effects = no
		inherit_icon = yes
		inherit_name = no
		trigger = {
			has_ethic = ethic_gestalt_consciousness
		}

			ship_weapon_damage = 0.10

		}

	}

		factor = 3000
	}
}
