country_event = {
	id = megaremove.2
	title = "megaremove.2.name"
	desc = "megaremove.2.desc"
	location = event_target:offending_toaster_system
	is_triggered_only = yes
	
	immediate = {
		random_system_within_border = {
			limit = {
				has_any_megastructure = yes
			}
			save_event_target_as = unwanted_toaster_system
		}
	}
	
	option = {
		name = "Pick a different toaster system."
		country_event = { id = megaremove.2 }
	}
	
	option = {
		name = "Yes. Remove all from system."
		event_target:unwanted_toaster_system = {
			every_system_megastructure = {
				delete_megastructure = THIS
			}
		}
	}
	option = {
		name = "Nevermind"
	}
}
