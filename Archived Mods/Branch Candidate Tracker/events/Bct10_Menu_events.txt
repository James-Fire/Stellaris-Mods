### Main Menu
country_event = {
	id = BctEvent.0001
	title = BctEvent.0001.name
	desc = BctEvent.0001.desc
	picture = @BctConstMenuPicture
	is_triggered_only = yes
	
	immediate = {
		set_country_flag = BctMenuOpened
	}
	
	after = {
		hidden_effect = {
			if = {
				limit = { has_country_flag = BctFlagSpMenu }
				remove_country_flag = BctFlagSpMenu
			} else = { country_event = { id = BctEvent.0001 } }
		}
	}
	
	option = {
		name = BctTextSeparator.name
		custom_tooltip = BctTextSeparator.tooltip
		allow = { hidden_trigger = { always = no } }
	}
	
	
	option = {
		name = BctEvent.0001.111.name
		custom_tooltip = BctEvent.0001.111.tooltip
		trigger = {
			NOR = {
				has_country_flag = BctFlagMarkAlsoNoCommercialPact
				is_criminal_syndicate = yes
			}
		}
		hidden_effect = { set_country_flag = BctFlagMarkAlsoNoCommercialPact }
	}
	option = {
		name = BctEvent.0001.112.name
		custom_tooltip = BctEvent.0001.112.tooltip
		trigger = {
			has_country_flag = BctFlagMarkAlsoNoCommercialPact
			NOT = { is_criminal_syndicate = yes }
		}
		hidden_effect = { remove_country_flag = BctFlagMarkAlsoNoCommercialPact }
	}
	
	option = {
		name = BctEvent.0001.121.name
		custom_tooltip = BctEvent.0001.121.tooltip
		trigger = { NOT = { has_country_flag = BctFlagNoMarkFreePlanet } }
		hidden_effect = { set_country_flag = BctFlagNoMarkFreePlanet }
	}
	option = {
		name = BctEvent.0001.122.name
		custom_tooltip = BctEvent.0001.122.tooltip
		trigger = { has_country_flag = BctFlagNoMarkFreePlanet }
		hidden_effect = { remove_country_flag = BctFlagNoMarkFreePlanet }
	}
	
	option = {
		name = BctEvent.0001.131.name
		custom_tooltip = BctEvent.0001.131.tooltip
		trigger = { NOT = { has_country_flag = BctFlagMarkPlanetWithForeignBranch } }
		hidden_effect = { set_country_flag = BctFlagMarkPlanetWithForeignBranch }
	}
	option = {
		name = BctEvent.0001.132.name
		custom_tooltip = BctEvent.0001.132.tooltip
		trigger = { has_country_flag = BctFlagMarkPlanetWithForeignBranch }
		hidden_effect = { remove_country_flag = BctFlagMarkPlanetWithForeignBranch }
	}
	
	option = {
		name = BctEvent.0001.141.name
		custom_tooltip = BctEvent.0001.141.tooltip
		trigger = { NOT = { has_country_flag = BctFlagPlayNotificationSound } }
		hidden_effect = {
			set_country_flag = BctFlagPlayNotificationSound
			play_sound = @BctConstNotificationSound
		}
	}
	option = {
		name = BctEvent.0001.142.name
		custom_tooltip = BctEvent.0001.142.tooltip
		trigger = { has_country_flag = BctFlagPlayNotificationSound }
		hidden_effect = { remove_country_flag = BctFlagPlayNotificationSound }
	}
	
	
	option = {
		name = BctTextSeparator.name
		custom_tooltip = BctTextSeparator.tooltip
		allow = { hidden_trigger = { always = no } }
	}
	
	
	option = {
		name = BctEvent.0001.210.name
		custom_tooltip = BctEvent.0001.210.tooltip
		hidden_effect = {
			set_country_flag = BctFlagSpMenu
			country_event = { id = BctEvent.0011 }
		}
	}
	
	option = {
		name = BctEvent.0001.220.name
		custom_tooltip = BctEvent.0001.220.tooltip
		hidden_effect = {
			set_country_flag = BctFlagSpMenu
			country_event = { id = BctEvent.0021 }
		}
	}
	
	option = {
		name = BctEvent.0001.230.name
		custom_tooltip = BctEvent.0001.230.tooltip
		hidden_effect = {
			set_country_flag = BctFlagSpMenu
			country_event = { id = BctEvent.0031 }
		}
	}
	
	option = {
		name = BctTextSeparator.name
		custom_tooltip = BctTextSeparator.tooltip
		allow = { hidden_trigger = { always = no } }
	}
	
	option = {
		name = BctTextEndEventChain.name
		custom_tooltip = BctTextEndEventChain.tooltip
		hidden_effect = {
			end_event_chain = BranchCandidateTracker_chain
			set_timed_country_flag = { flag = BctFlagForbidEventChain days = 7200 }
			set_country_flag = BctFlagSpMenu
		}
	}
	
	option = {
		name = BctTextSeparator.name
		custom_tooltip = BctTextSeparator.tooltip
		allow = { hidden_trigger = { always = no } }
	}
	
	option = {
		name = BctTextExitMenu.name
		custom_tooltip = BctTextExitMenu.tooltip
		hidden_effect = {
			set_country_flag = BctFlagSpMenu
			remove_country_flag = BctMenuOpened
			country_event = { id = BctEvent.1002 }
			country_event = { id = BctEvent.1101 }
		}
	}
}
