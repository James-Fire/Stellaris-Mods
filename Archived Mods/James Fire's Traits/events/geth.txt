# AI Geth empires do not fare well. I compensated a bit for the AI's bad priorities but it's better to keep the trait solely for human players


namespace = geth

event = {
	id = geth.0
	hide_window = yes
    is_triggered_only = yes	
    immediate = {
		every_country = {
			limit = {
				has_trait = trait_robodesign_geth
			}
			every_owned_planet = {
				limit = {
					any_owned_pop = {
						pop_has_trait = trait_robodesign_geth
					}
				}
				set_variable = { which = geth_pops value = 0 }
				every_owned_pop = {
					limit = { pop_has_trait = trait_robodesign_geth }
					PREV = { change_variable = { which = geth_pops value = 1 } }
				}
				change_variable = { which = geth_pops value = -5 }
				add_modifier = {
					modifier = geth
					multiplier = geth_pops
				}
			}
		}
	}
}
