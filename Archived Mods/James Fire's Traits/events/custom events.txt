namespace = jftraits

country_event = {
	id = jftraits.4000
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {	
		has_ascension_perk = ap_synthetic_age
		NOT = {
			has_country_flag = has_robo_engineering
		}
	}
	
	immediate = {		
		if = {
			limit = {
				has_tech_option = tech_nano_robo_engineering
			}
			add_tech_progress = {
				tech = tech_nano_robo_engineering
				progress = 0.25
			}
		}
		else = {
			add_research_option = tech_nano_robo_engineering
		}
		set_country_flag = has_robo_engineering
	}
}
