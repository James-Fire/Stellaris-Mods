building_bioreactor = {
	icon = "building_bio_reactor"
	base_buildtime = 360
	base_cap_amount = 1
	
	category = research
	
	prerequisites = {
		"tech_bioreactor"
	}
	
	destroy_trigger = {
		OR = {
			has_building = building_clinic
			has_building = building_hospital
		}
		exists = owner
		owner = { is_gestalt = yes }
	}

	potential = {
		NOT = { has_modifier = resort_colony }	
		NOT = { has_modifier = slave_colony }
		exists = owner
		owner = { is_gestalt = no }
		NOR = {
			has_building = building_clinic
			has_building = building_hospital
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_bio_reactor_technician_add = 2
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			alloys = 300
		}
		upkeep = {
			volatile_motes = 2
			exotic_gases = 2
			food = 10
			energy = 10
		}
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_regular_empire = yes } 
		}
		text = job_bio_reactor_technician_effect_desc
	}

	upgrades = {
		building_bioreactor_2
	}

	planet_modifier = {
		planet_pop_assembly_organic_add = 2
	}

	ai_weight = {
		weight = 20
	}
}
