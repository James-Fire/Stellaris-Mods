  entity = ""

  portrait = "GFX_megastructure_construction_background"
  place_entity_on_planet_plane = yes
  entity_offset = { x = -7 y = -7 }
  build_time = 200
  resources = {
		category = megastructures
		cost = {
			alloys = 200
			influence = 2
		}
	}
  construction_blocks_others = no

  potential = {
    is_ai=no
    has_country_flag = use_a_mega
  }

  possible = {}

  placement_rules = {
    planet_possible = {
      custom_tooltip = {
        fail_text = "requires_no_anomaly"
        NOT = { has_anomaly = yes }
      }
    }
  }

  ai_weight = {
    modifier = {
      weight = 0
    }
  }

  on_build_start = {}
  on_build_cancel = {}

  on_build_complete = {
    remove_megastructure = fromfrom

      set_planet_flag = is_choosing_psize
    }
    from = {
      country_event = { id = change_paradise_type.112 }
      country_event = { id = change_paradise_type.100 }
    }
  }
}
