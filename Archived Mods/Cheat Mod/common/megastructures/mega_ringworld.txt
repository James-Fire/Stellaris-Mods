modify_ringworld = {
  entity = ""
  construction_entity = "construction_platform_entity"
  portrait = "GFX_megastructure_construction_background"
  place_entity_on_planet_plane = yes
  entity_offset = { x = -7 y = -7 }
  build_time = 100
  resources = {
		category = megastructures
		cost = {
			alloys = 100
			influence = 1
		}
	}
  construction_blocks_others = no

  potential = {
    is_ai=no
    has_country_flag = use_a_mega
  }

  possible = {}

  placement_rules = {
    planet_possible = {
      custom_tooltip = {
        fail_text = "requires_no_anomaly"
        NOT = { has_anomaly = yes }
      }
      custom_tooltip = {
        fail_text = "requires_ringworld_not_habitable"
        OR = {
          is_planet_class = pc_ringworld_seam
          is_planet_class = pc_ringworld_tech
          is_planet_class = pc_ringworld_habitable_damaged
          is_planet_class = pc_ringworld_habitable
        }
      }
    }
  }

  ai_weight = {
    modifier = {
      weight = 0
    }
  }

  on_build_start = {}
  on_build_cancel = {}

  on_build_complete = {
    remove_megastructure = fromfrom
    fromfrom.planet = {
      if = {
        limit = {
          NOR = {
            is_planet_class = pc_ringworld_habitable
            is_planet_class = pc_ringworld_habitable_damaged
          }
        }
        change_planet_size = 15
      }
      if = {
        limit = {
          NOT = {
            is_planet_class = pc_ringworld_habitable
          }
        }
        change_pc = pc_ringworld_habitable
        aai_reroll_planet = yes
      }
      kill_all_blockers = yes
      set_planet_flag = graphical_style_changing
    }
    from = {
      country_event = { id = change_paradise_type.112 }
      country_event = { id = change_paradise_type.103 }
    }
  }
}
