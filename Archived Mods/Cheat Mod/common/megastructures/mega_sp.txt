				has_megastructure = gateway_restored
				has_megastructure = gateway_final
				has_megastructure = player_gateway_alpha
			}
		}
	}
	placement_rules = {}
	build_time = 100
	resources = {
		category = megastructures
		cost = {
			alloys = 150
			influence = 100
		}
	}

		factor = 0
	}
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		from = {
			if = {
				limit = { NOT = { has_technology = tech_gateway_activation } }
				give_technology = { message = no tech = tech_gateway_activation }
			}
			country_event = { id = apoc.14 }
		}
		activate_gateway = fromfrom
		if = {
			limit = { num_active_gateways = 1 }
			random_megastructure = {
				limit = { is_megastructure_type = gateway_ruined }
				upgrade_megastructure_to = gateway_restored
				finish_upgrade = yes
			}
		}
		from = {
			country_event = { id = apoc.13 }
			country_event = { id = apoc.11 }
		}
	}
}
