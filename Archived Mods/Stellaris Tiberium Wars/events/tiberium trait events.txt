#Monthly event manager for trait related events
country_event = {
	id = tiberium_traits.0
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		every_country = {
			limit = {
				country_has_tiberium = yes
			}
			fire_on_action = { on_action = tiberium_traits_cycle }
		}
	}
}


#Recheck whether the country has tib stuff, for other events.
country_event = {
	id = tiberium_traits.1000
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		if = {
			limit = {
				OR = {
					country_has_tiberium = yes
					any_owned_species = {
						OR = {
							has_trait = trait_tiberium_infected
							has_trait = trait_tiberium_infested
						}
					}
				}
			}
			set_country_flag = has_tiberium
		}
	}
}

							}

							}
						}
					}
				}
			}
