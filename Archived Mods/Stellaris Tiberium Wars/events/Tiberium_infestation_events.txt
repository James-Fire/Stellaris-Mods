planet_event = {
	id = tiberium_infest.0
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		OR = {
			is_planet_class = pc_habitable_tiberium
			is_planet_class = pc_uninhabitable_tiberium
			has_deposit = d_tiberium_asteroid
			has_planet_flag = tiberium_asteroid_collision
		}
		NOT = { has_planet_flag = tiberium_initial_infest_delay }
	}
	
	immediate = {
		remove_planet_flag = tiberium_grew
		remove_planet_flag = tiberium_shrunk
		if = {
			limit = {
				NOT = { has_planet_flag = tiberium_variable_setup_done }
			}
			set_tiberium_variables = yes
		}
		if = {
			limit = {
				NOT = {
					ROOT = { has_planet_flag = tiberium_depleted }
				}
				OR = {
					has_deposit = d_tiberium_asteroid
					has_planet_flag = tiberium_asteroid_collision
					is_planet_class = pc_habitable_tiberium
					is_planet_class = pc_uninhabitable_tiberium
				}
			}
			planet_event = { id = tiberium_infest.1 }
		}
	}
}
