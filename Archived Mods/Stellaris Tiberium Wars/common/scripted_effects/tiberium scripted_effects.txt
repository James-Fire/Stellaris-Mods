set_tiberium_variables = {
	set_planet_flag = tiberium_variable_setup_done
	set_variable = { which = tiberium_takeover_count value = 0 }
	set_variable = { which = tiberium_total_count value = 0 }
	set_variable = { which = tiberium_total_cap value = trigger:planet_size }
	multiply_variable = { which = tiberium_total_cap value = 6 }
	add_modifier = {
		modifier = tiberium_infestation_status
		days = -1
	}
	set_timed_planet_flag = {
		flag = tiberium_initial_infest_delay
		months = 6
	}
}

tib_count_pops_takeover = {
	every_owned_pop = {
		limit = { has_trait = trait_tiberium_mutated }
		ROOT = { change_variable = { which = tiberium_takeover_count value = 2 } }
	}
	every_owned_pop = {
		limit = {
			OR = {
				has_trait = trait_tiberium_infected
				has_trait = trait_tiberium_infested
			}
		}
		ROOT = { change_variable = { which = tiberium_takeover_count value = 1 } }
	}
	every_owned_pop = {
		limit = {
			has_job = tiberium_researcher
			NOT = {
				ROOT = { has_planet_flag = tiberium_depleted }
			}
		}
		ROOT = { change_variable = { which = tiberium_takeover_count value = -1 } }
	}
	every_owned_pop = {
		limit = { has_job = green_tiberium_harvester }
		ROOT = { change_variable = { which = tiberium_takeover_count value = -1 } }
	}
	every_owned_pop = {
		limit = { has_job = blue_tiberium_harvester }
		ROOT = { change_variable = { which = tiberium_takeover_count value = -1 } }
	}
}
