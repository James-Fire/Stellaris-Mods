tiberium_green_refinery = {

		}

			alloys = 1
		}

			trigger = {
				exists = owner
				owner = { has_technology = tech_tiberium_refining }
				owner = { NOT = { has_deficit = sr_tiberium_green } }

			energy = 1
		}
		produces = {
			trigger = {
				exists = owner
				owner = { has_technology = tech_tiberium_refining_2 }

			}
			minerals = 4
			alloys = 1
		}
		upkeep = {
			trigger = {
				exists = owner

			exists = owner

				is_gestalt = no

		potential = {
			exists = owner
			owner = { 
				is_gestalt = yes
			}
		}
		modifier = {
			job_green_tiberium_refiner_gestalt_add = 1
		}

	prerequisites = {
		"tech_tiberium_green"
	}

		"tiberium_green_refinery_2"

		weight = 0

			weight = 5
