building_tiberium_refinery = {
	icon = "building_tiberium_refinery_green"
	base_buildtime = 480
	base_cap_amount = 2
	
	owner_type = corporate
	
	potential = {
		has_branch_office = yes
		branch_office_owner = {
			is_criminal_syndicate = no
			has_technology = tech_tiberium_green
		}
		
		OR = {
			NOT = { 
				branch_office_owner = { has_policy_flag = tiberium_safe }
			}
			AND = {
				OR = {
					branch_office_owner = { has_policy_flag = tiberium_safe }
					branch_office_owner = { has_policy_flag = tiberium_priority_safe }
				}
				OR = {
					is_planet_class = pc_habitable_tiberium
					is_planet_class = pc_uninhabitable_tiberium
				}
			}
		}
	}
	
	planet_modifier = {
		job_green_tiberium_refiner_add = 1
		job_blue_tiberium_refiner_add = 1
	}

	resources = {
		category = planet_branch_office_buildings
		cost = {
			minerals = 300
		}
		produces = {
			trigger = {
				exists = owner
				owner = { NOT = { has_deficit = sr_tiberium_green } }
			}
			minerals = 20
			alloys = 5
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = { NOT = { has_deficit = sr_tiberium_green } }
			}
			sr_tiberium_green = 5 #4/1
			alloys = 0.5
			energy = 2
		}
		produces = {
			trigger = {
				exists = owner
				owner = { NOT = { has_deficit = sr_tiberium_blue } }
			}
			minerals = 10
			alloys = 25
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = { NOT = { has_deficit = sr_tiberium_blue } }
			}
			sr_tiberium_blue = 5 #2/5
			alloys = 0.5
			energy = 2
		}
	}
	
	ai_weight = {
		weight = 1
	}
}
