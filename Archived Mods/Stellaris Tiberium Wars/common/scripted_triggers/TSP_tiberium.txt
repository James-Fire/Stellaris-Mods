	OR = {
		has_trait = trait_mechanical
		has_trait = trait_machine_unit
		has_trait = trait_cybernetic
		species = { is_archetype = ROBOT }
		species = { is_archetype = MACHINE }
	}
	NOR = { # All robots can hypothetically mine Tiberium, but Tiberium eats robots, so only reckless cost cutting would use tier one robots.
		owner = { has_policy_flag = "tiberium_safe" }
		owner = { has_policy_flag = "tiberium_priority_safe" }
		owner = { has_policy_flag = "tiberium_balanced_safe" }
	}
}
