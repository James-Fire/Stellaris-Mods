	min_for_deposits_on_planet = 0
	max_for_deposits_on_planet = 999

		NOT = {
			is_planet_class = pc_habitable_tiberium

			has_deposit = d_tiberium_asteroid
			has_planet_flag = tiberium_asteroid_collision
		}
	}

			is_planet_class = pc_habitable_tiberium

		}

			energy = 2
		}
	}	
	
	planet_modifier = {
		planet_housing_add = 2
	}

				is_gestalt = no
			}

		}
