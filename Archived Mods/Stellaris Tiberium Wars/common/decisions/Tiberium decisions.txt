infect_planet_green = {
	sound = event_administrative_work
	icon = decision_resources
	
	prerequisites = {
		"tech_tiberium_green"		
	}
	
	resources = {
		category = decisions
		cost = {
			sr_tiberium_green = 10
		}
	}
	
	potential = {
		NOR = { 
			is_planet_class = pc_uninhabitable_tiberium
			is_planet_class = pc_frozen
			is_planet_class = pc_barren_cold
		}
		OR = {
			is_planet_class = pc_barren
			is_planet_class = pc_toxic
			is_planet_class = pc_molten
			has_deposit = d_minerals_2
			has_deposit = d_minerals_3
			has_deposit = d_minerals_4
			has_deposit = d_minerals_5
			has_deposit = d_minerals_6
			has_deposit = d_minerals_7
			has_deposit = d_minerals_8
			has_deposit = d_minerals_9
			has_deposit = d_minerals_10
		}
	}
	
	allow = {
		always = yes
	}	
	
	effect = {
		custom_tooltip = "seed_green_tib"
		hidden_effect = {
			tib_convert_planet = yes
			while = {
				count = 2
				add_deposit = d_green_tiberium_field
			}
		}
	}
	
	ai_weight = {
		weight = 1
	}
}

infect_habitable_planet_green = {
	sound = event_administrative_work
	icon = decision_resources
	
	prerequisites = {
		"tech_tiberium_green"		
	}
	
	resources = {
		category = decisions
		cost = {
			sr_tiberium_green = 10
		}
	}
	
	potential = {
		NOR = { 
			is_planet_class = pc_habitable_tiberium
			is_planet_class = pc_frozen
			has_planet_flag = tiberium_asteroid_collision
		}
		OR = {
			habitable_planet = yes
			is_planet_class = pc_city
			is_planet_class = pc_nuked
			is_planet_class = pc_hive
			is_planet_class = pc_machine
		}
	}
	
	allow = {
		always = yes
	}	
	
	effect = {
		custom_tooltip = "seed_green_tib_colony"
		hidden_effect = {
			tib_convert_planet = yes
			while = {
				count = 1
				add_deposit = b_red_zone
			}
			while = {
				count = 2
				add_deposit = b_yellow_zone
			}
			while = {
				count = 50
				limit = {
					num_free_districts = { type = any value > 0 }
				}
				remove_last_built_district = yes
				ROOT = { add_district = district_city }
				add_deposit = b_blue_zone
			}
			remove_last_built_district = yes
			every_deposit = {
				limit = { is_deposit_type = b_blue_zone }
				remove_deposit = yes
				ROOT = { add_deposit = d_blue_zone }
			}
			every_deposit = {
				limit = { is_deposit_type = b_yellow_zone }
				remove_deposit = yes
				ROOT = { add_deposit = d_yellow_zone }
			}
			every_deposit = {
				limit = { is_deposit_type = b_red_zone }
				remove_deposit = yes
				ROOT = { add_deposit = d_red_zone }
			}
			add_deposit = d_blue_zone
		}
	}
	
	
	ai_weight = {
		weight = 1
	}
}

tiberium_clean_pops = {
	sound = event_administrative_work
	icon = decision_resources
	
	prerequisites = {
		"tech_tiberium_sonics"
	}
	
	resources = {
		category = decisions
		cost = {
			energy = 100
		}
	}
	
	potential = {
		any_owned_pop = {
			OR = {
				has_trait = trait_tiberium_infected
				has_pop_flag = tiberium_infected
			}
		}
	}
	
	allow = {
		always = yes
	}	
	
	effect = {
		hidden_effect = {
			every_owned_pop = {
				remove_pop_flag = tiberium_infected
				limit = {
					has_trait = trait_tiberium_infected
				}
				modify_species = {
					species = this
					remove_trait = trait_tiberium_infected
				}
			}
		}
	}
	
	ai_weight = {
		weight = 1
	}
}
