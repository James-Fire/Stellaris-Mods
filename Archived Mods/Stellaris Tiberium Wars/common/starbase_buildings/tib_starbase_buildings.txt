			sr_tiberium_green = 1
			energy = 1
		}
		produces = {
			trigger = {
				exists = owner
				owner = { has_technology = tech_tiberium_refining_2 }
				owner = { NOT = { has_deficit = sr_tiberium_green } }
			}
			minerals = 4
			alloys = 1
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = { has_technology = tech_tiberium_refining_2 }
				owner = { NOT = { has_deficit = sr_tiberium_green } }
			}
			sr_tiberium_green = 1
			energy = 1
		}
	}
	
	potential = {
		exists = owner
		owner = { has_technology = tech_tiberium_green }
	}

	ai_build_outside_chokepoint = yes
	ai_weight = {
		weight = 0
		modifier = {
			weight = 5
			owner = {
				AND = {
					has_monthly_income = {
						resource = sr_tiberium_green
						value > 5
					}
					OR = {
						has_monthly_income = {
							resource = alloys
							value < 20
						}
						has_monthly_income = {
							resource = minerals
							value < 20
						}
					}
				}
			}
		}
	}
