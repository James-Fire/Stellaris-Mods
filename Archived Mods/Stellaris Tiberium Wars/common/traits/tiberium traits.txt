trait_tiberium_infected = {
	cost = 0
	initial = no
	randomized = no
	species_potential_add = {
		hidden_trigger = { exists = from }
		from = {
			any_owned_species = {
				has_trait = trait_tiberium_infected
			}
		}
	}
	species_possible_add = {
		always = yes
	}
	species_possible_remove = {
		always = no
	}
	potential_crossbreeding_chance = 1.0	# 1.0 = 100% chance of being considered for new traits when forming half-species. does not guarantee the trait will be added if it costs points.
	allowed_archetypes = { BIOLOGICAL LITHOID ROBOT MACHINE }
	modifier = {
		pop_happiness = -0.10
	}
	slave_cost = {
		energy = 0
	}
}
