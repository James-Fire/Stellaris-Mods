planet_event = {
	id = bureaucratic.3
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		every_deposit = {
			if = {
				limit = {
					OR = {
						is_deposit_type = d_ship_shelter_modifiers
						is_deposit_type = d_planetary_administration_modifiers
						is_deposit_type = d_planetary_capital_modifiers
						is_deposit_type = d_system_capital_modifiers
						is_deposit_type = d_habitat_modifiers
						is_deposit_type = d_major_habitat_modifiers
					}
				}
				remove_deposit = yes
			}
		}
		if = {
			limit = {
				has_level_1_capital = yes
				is_under_colonization = no
			}
			add_deposit = d_ship_shelter_modifiers
		}
		if = {
			limit = {
				has_level_2_capital = yes
				is_under_colonization = no
			}
			add_deposit = d_planetary_administration_modifiers
		}
		if = {
			limit = {
				has_level_3_capital = yes
				is_under_colonization = no
			}
			add_deposit = d_planetary_capital_modifiers
		}
		if = {
			limit = {
				has_level_4_capital = yes
				is_under_colonization = no
			}
			add_deposit = d_system_capital_modifiers
		}
		if = {
			limit = {
				has_hab_1_capital = yes
				is_under_colonization = no
			}
			add_deposit = d_habitat_modifiers
		}
		if = {
			limit = {
				has_hab_2_capital = yes
				is_under_colonization = no
			}
			add_deposit = d_major_habitat_modifiers
		}
	}
}
