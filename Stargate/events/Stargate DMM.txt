event = {
	id = stargate_dmm.1
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		dmm_register_mod = {
			DMM_NAME = stargate_menu_name
			DMM_FLAG = stargate_menu_name_flag
		}
	}
}

	is_triggered_only = yes

	trigger = {
		from = {
			has_leader_flag = stargate_menu_name_flag
		}
	}
	immediate = {
		country_event = { id = stargate_menu.1 }
	}
}
