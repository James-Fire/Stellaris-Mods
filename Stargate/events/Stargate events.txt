		#set_variable = { which = stargate_research_connections_ex_con value = 0 }

		every_owned_colony = {
			limit = {
				num_pops > 0
			}
			set_stargate_variables = yes
			PREV = { change_variable = { which = stargate_owned_colonys value = 1 } }
		}
		every_owned_colony = {
			limit = {
				has_deposit = d_stargate
				num_pops > 0
			}
			PREV = { change_variable = { which = stargatecount value = 1 } }

				limit = {
					has_global_flag = stargate_logging
				}
				log = "Valid source planet found on [This.GetName]"
			}
		}
		#stargate_primitive_connections = yes
	}
	
	
	after = {
		set_variable = { which = stargate_fraction value = stargatecount }
		set_variable = { which = stargate_malus_count value = stargatecount }
		divide_variable = { which = stargate_fraction value = stargate_owned_colonys }
		set_variable = { which = accessible_internal_civilian_stargatecount value = stargatecount }
		set_variable = { which = accessible_internal_military_stargatecount value = stargatecount }

			log = "Research Connections Start for [This.GetName] [This.accessible_external_research_stargatecount]"
		}
		if = {
			limit = {
				NOT = {
					is_country_type = primitive
				}
			}
			country_event = { id = stargate.2 days = 1 }
		}
	}

country_event = {
	id = stargate.2
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		every_subject = {
			limit = {
				NOT = {
					is_in_federation_with = ROOT
				}
				has_stargate = yes
				ROOT = { is_overlord = yes }
			}
			stargate_connection_opinion_set = yes
			transfer_stargate_connection_count = yes
			if = {
				limit = {
					has_global_flag = stargate_logging
				}
				log = "Stargates Subject [This.GetName] [This.stargatecount]"
			}
			ROOT = {
				stargate_connection_opinion_set = yes
				transfer_stargate_connection_count = yes
				if = {
					limit = {
						has_global_flag = stargate_logging
					}
					log = "Subject External Military Connections after [PREV.GetName] for [This.GetName] [This.accessible_external_military_stargatecount]"
					log = "Subject External Civilian Connections after [PREV.GetName] for [This.GetName] [This.accessible_external_civilian_stargatecount]"
					log = "Subject Research Connections after [PREV.GetName] for [This.GetName] [This.accessible_external_research_stargatecount]"
				}
			}
		}
	}
	after = {
		if = {
			limit = {
				has_global_flag = stargate_logging
			}
			log = "Stargates Host Before Allies [This.GetName] [This.stargatecount]"
		}
		every_country = {
			limit = {
				NOT = { is_same_value = prev }
				NOT = {
					is_country_type = primitive
				}
				has_communications = ROOT
				has_stargate = yes
			}
			remove_opinion_modifier = {
				who = root
				modifier = opinion_stargate_access
			}
		}
		every_country = {
			limit = {
				NOT = { is_same_value = prev }
				NOT = {
					is_country_type = primitive
				}
				has_communications = ROOT
				has_stargate = yes
			}
			if = {
				limit = {
					has_global_flag = stargate_logging
				}
				log = "Stargates Military Ally [This.GetName] [This.stargatecount]"
			}
			stargate_connection_opinion_set = yes
			transfer_stargate_connection_count = yes
			ROOT = {
				stargate_connection_opinion_set = yes
				transfer_stargate_connection_count = yes
			}
		}
		reset_stargate_resettle = yes
		#accessible_internal_military_stargatecount
		#accessible_internal_civilian_stargatecount
		#stargate_connection_mult
		country_event = { id = stargate.3 days = 1 }
	}
}
