has_stargate = {
	OR = {
		has_technology = tech_stargate
		any_owned_colony = {
			has_deposit = d_stargate
		}
	}
}

can_have_seed_stargate = {
	OR = {
		#is_terraforming_candidate = yes #Guilli's compat
		has_modifier = terraforming_candidate
		is_colonizable = yes
		is_colony = yes
	}
	NOR = {
		has_deposit = d_stargate
		has_carrier_flag = stargate_check
		has_carrier_flag = has_stargate
		is_planet_class = pc_gas_giant
	}
}

has_civilian_connection = {
	has_communications = ROOT
	OR = {
		AND = {
			has_policy_flag = stargate_civilian_open
			NOT = { 
				has_closed_borders = ROOT
			}
			ROOT = {
				has_policy_flag = stargate_civilian_open
			}
		}
		AND = {
			has_commercial_pact = ROOT
			OR = {
				has_policy_flag = stargate_civilian_open
				has_policy_flag = stargate_civilian_economy
			}
			ROOT = {
				OR = {
					has_policy_flag = stargate_civilian_open
					has_policy_flag = stargate_civilian_economy
				}
			}
		}
		AND = {
			has_research_agreement = ROOT
			OR = {
				has_policy_flag = stargate_civilian_open
				has_policy_flag = stargate_civilian_economy
				has_policy_flag = stargate_civilian_science
			}
			ROOT = {
				OR = {
					has_policy_flag = stargate_civilian_open
					has_policy_flag = stargate_civilian_economy
					has_policy_flag = stargate_civilian_science
				}
			}
		}
		AND = {
			has_association_status = ROOT
			OR = {
				has_policy_flag = stargate_civilian_open
				has_policy_flag = stargate_civilian_economy
				has_policy_flag = stargate_civilian_science
				has_policy_flag = stargate_civilian_allies
			}
			ROOT = {
				OR = {
					has_policy_flag = stargate_civilian_open
					has_policy_flag = stargate_civilian_economy
					has_policy_flag = stargate_civilian_science
					has_policy_flag = stargate_civilian_allies
				}
			}
		}
		AND = {
			is_in_federation_with = ROOT
			OR = {
				has_policy_flag = stargate_civilian_open
				has_policy_flag = stargate_civilian_economy
				has_policy_flag = stargate_civilian_science
				has_policy_flag = stargate_civilian_allies
				has_policy_flag = stargate_civilian_fed
			}
			ROOT = {
				OR = {
					has_policy_flag = stargate_civilian_open
					has_policy_flag = stargate_civilian_economy
					has_policy_flag = stargate_civilian_science
					has_policy_flag = stargate_civilian_allies
					has_policy_flag = stargate_civilian_fed
				}
			}
		}
	}

has_int_econ_con_space = {
	check_variable = { which = stargate_connections value <= 90 }
	check_variable = { which = accessible_internal_civilian_stargatecount value > 1 }
	OR = {
		count_deposit = { limit = { is_deposit_type = d_stargate_internal_civilian_network } count < int_civ_stargatecount_cap }
		check_variable = { which = int_civ_stargatecount_cap value = 0 }
	}
	AND = {
		NOT = { has_carrier_flag = int_econ_con_disabled }
		NAND = {
			owner = { has_country_flag = int_econ_con_disabled }
			has_carrier_flag = int_econ_con_default
		}
	}
}
