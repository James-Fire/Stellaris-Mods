planet_stargate_mil_con_int_increase_cap = {
	potential = { }
	allow = {
		custom_tooltip = {
			fail_text = country_caps_used
			has_planet_flag = stargate_local_caps
		}
		custom_tooltip = {
			fail_text = at_max_connections
			NOT = {
				check_variable = { which = stargate_military_connections_int_cap value >= 90 }
				check_variable = { which = stargate_connections_cap value >= 90 }
			}
		}
		custom_tooltip = {
			fail_text = connection_type_disabled
			NOT = { has_planet_flag = stargate_military_connections_int_off }
		}
	}
	effect = {
		custom_tooltip = "stargate_connection_cap_increase"
		hidden_effect = {
			if = {
				limit = {
					has_planet_flag = stargate_military_connections_int_cap_1
					check_variable = { which = stargate_connections_cap value < 90 }
				}
				change_variable = { which = stargate_military_connections_int_cap value = 1 }
				change_variable = { which = stargate_connections_cap value = 1 }
			}
			else_if = {
				limit = {
					has_planet_flag = stargate_military_connections_int_cap_2
					check_variable = { which = stargate_connections_cap value < 89 }
				}
				change_variable = { which = stargate_military_connections_int_cap value = 2 }
				change_variable = { which = stargate_connections_cap value = 2 }
			}
			else_if = {
				limit = {
					has_planet_flag = stargate_military_connections_int_cap_3
					check_variable = { which = stargate_connections_cap value < 88 }
				}
				change_variable = { which = stargate_military_connections_int_cap value = 3 }
				change_variable = { which = stargate_connections_cap value = 3 }
			}
			else_if = {
				limit = {
					has_planet_flag = stargate_military_connections_int_cap_4
					check_variable = { which = stargate_connections_cap value < 87 }
				}
				change_variable = { which = stargate_military_connections_int_cap value = 4 }
				change_variable = { which = stargate_connections_cap value = 4 }
			}
			else_if = {
				limit = {
					has_planet_flag = stargate_military_connections_int_cap_5
					check_variable = { which = stargate_connections_cap value < 86 }
				}
				change_variable = { which = stargate_military_connections_int_cap value = 5 }
				change_variable = { which = stargate_connections_cap value = 5 }
			}
		}
	}
}
