	set_variable = { which = stargate_connection_mult value = stargate_connections }
	set_variable = { which = stargate_connection_multi value = stargate_connections }
	if = {
		limit = {
			has_global_flag = stargate_logging
		}
		log = "Stargate Connection mult Step 1 for \\[This.GetName] \\[This.stargate_connection_mult]"
	}
	#(a*a)/160+1.9a-0.9
	calc_connections_upkeep_part_2 = yes
}

check_galaxy_stargate_deposit = {
	every_system = {
		limit = {
			any_system_planet = {
				has_planet_flag = has_stargate
				NOT = {
					has_deposit = d_stargate
				}
			}
		}
		every_system_planet = {
			limit = {
				has_planet_flag = has_stargate
				NOT = {
					has_deposit = d_stargate
				}
			}
			add_deposit = d_stargate
		}
	}
}

transfer_stargate_connection_count = {
	if = {
		limit = {
			has_civilian_connection = yes
		}
		change_variable = { which = accessible_external_civilian_stargatecount value = PREV.stargatecount }
	}
	if = {
		limit = {
			has_military_connection = yes		
		}
		change_variable = { which = accessible_external_military_stargatecount value = PREV.stargatecount }
	}
	if = {
		limit = {
			has_research_connection = yes
		}
		change_variable = { which = accessible_external_research_stargatecount value = PREV.stargatecount }
	}
}

stargate_connection_opinion_set = {
	set_variable = { which = opinion_stargatecount value = PREV.stargatecount }
	if = {
		limit = {
			has_global_flag = stargate_logging
		}
		log = "Diplo Stargates Subject \\[This.GetName] \\[This.diplo_stargatecount] for \\[PREV.GetName]"
	}
	while = {
		limit = {
			check_variable = { which = opinion_stargatecount value > 0 }
		}
		change_variable = { which = opinion_stargatecount value = -1 }
		add_opinion_modifier = {
			who = prev
			modifier = opinion_stargate_access
		}
	}
}

#Country Scope
stargate_primitive_connections = {
	every_planet_within_border = {
		limit = {
			OR = {
				is_colonizable = yes
				is_colony = yes
			}
			has_deposit = d_stargate
			has_observation_outpost = yes
			observation_outpost = { has_mission = primitive_tribute }
			exists = owner
			owner = {
				is_country_type = primitive
				is_primitive = yes
				OR = {
					has_country_flag = stone_age
					has_country_flag = bronze_age
					has_country_flag = iron_age
					has_country_flag = late_medieval_age
					has_country_flag = renaissance_age
					has_country_flag = steam_age
					has_country_flag = industrial_age
					has_country_flag = machine_age
					has_country_flag = atomic_age
					has_country_flag = early_space_age
				}
			}
		}
		if = {
			limit = {
				has_global_flag = stargate_logging
			}
			log = "Valid primitive planet found for \\[PREV.GetName]"
		}
		ROOT = { change_variable = { which = accessible_civilian_stargatecount value = 1 } }
		ROOT = { change_variable = { which = accessible_military_stargatecount value = 1 } }
	}
}
